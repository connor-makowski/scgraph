 152    def dijkstra(
 153        graph: list[dict], origin_id: int, destination_id: int
 154    ) -> dict:
 155        """
 156        Function:
 157
 158        - Identify the shortest path between two nodes in a sparse network graph using a modified dijkstra algorithm
 159            - Modifications allow for a sparse distance matrix to be used instead of a dense distance matrix
 160            - This can dramatically reduce the memory and compute requirements of the algorithm
 161            - This algorithm runs in O(n^2) time where n is the number of nodes in the graph
 162        - Return a dictionary of various path information including:
 163            - `id_path`: A list of node ids in the order they are visited
 164            - `path`: A list of node dictionaries (lat + long) in the order they are visited
 165
 166        Required Arguments:
 167
 168        - `graph`:
 169            - Type: list of dictionaries
 170            - See: https://connor-makowski.github.io/scgraph/scgraph/core.html#GeoGraph
 171        - `origin_id`
 172            - Type: int
 173            - What: The id of the origin node from the graph dictionary to start the shortest path from
 174        - `destination_id`
 175            - Type: int
 176            - What: The id of the destination node from the graph dictionary to end the shortest path at
 177
 178        Optional Arguments:
 179
 180        - None
 181        """
 182        Graph.input_check(
 183            graph=graph, origin_id=origin_id, destination_id=destination_id
 184        )
 185        distance_matrix = [float("inf") for i in graph]
 186        branch_tip_distances = [float("inf") for i in graph]
 187        predecessor = [None for i in graph]
 188
 189        distance_matrix[origin_id] = 0
 190        branch_tip_distances[origin_id] = 0
 191
 192        while True:
 193            current_distance = min(branch_tip_distances)
 194            if current_distance == float("inf"):
 195                raise Exception(
 196                    "Something went wrong, the origin and destination nodes are not connected."
 197                )
 198            current_id = branch_tip_distances.index(current_distance)
 199            branch_tip_distances[current_id] = float("inf")
 200            if current_id == destination_id:
 201                break
 202            for connected_id, connected_distance in graph[current_id].items():
 203                possible_distance = current_distance + connected_distance
 204                if possible_distance < distance_matrix[connected_id]:
 205                    distance_matrix[connected_id] = possible_distance
 206                    predecessor[connected_id] = current_id
 207                    branch_tip_distances[connected_id] = possible_distance
 208
 209        output_path = [current_id]
 210        while predecessor[current_id] is not None:
 211            current_id = predecessor[current_id]
 212            output_path.append(current_id)
 213
 214        output_path.reverse()
 215
 216        return {
 217            "path": output_path,
 218            "length": hard_round(4, distance_matrix[destination_id]),
 219        }

 225        """
 226        Function:
 227
 228        - Identify the shortest path between two nodes in a sparse network graph using Makowski's modified Dijkstra algorithm

 231                - Open leaves are nodes that have not been visited yet but are adjacent to other visited nodes
 232            - This can dramatically reduce the memory and compute requirements of the algorithm
 233            - For particularly sparse graphs, this algorithm runs close to O(n log n) time
 234                - Where n is the number of nodes in the graph
 235            - For dense graphs, this algorithm runs closer to O(n^2) time (similar to the standard Dijkstra algorithm)
 236                - Where n is the number of nodes in the graph
 237        - Return a dictionary of various path information including:
 238            - `id_path`: A list of node ids in the order they are visited
 239            - `path`: A list of node dictionaries (lat + long) in the order they are visited
 240
 241        Required Arguments:
 242
 243        - `graph`:
 244            - Type: list of dictionaries
 245            - See: https://connor-makowski.github.io/scgraph/scgraph/core.html#GeoGraph
 246        - `origin_id`
 247            - Type: int
 248            - What: The id of the origin node from the graph dictionary to start the shortest path from
 249        - `destination_id`
 250            - Type: int
 251            - What: The id of the destination node from the graph dictionary to end the shortest path at
 252
 253        Optional Arguments:
 254
 255        - None
 256        """
 257        Graph.input_check(
 258            graph=graph, origin_id=origin_id, destination_id=destination_id
 259        )
 260        distance_matrix = [float("inf") for i in graph]
 261        open_leaves = {}
 262        predecessor = [None for i in graph]
 263
 264        distance_matrix[origin_id] = 0
 265        open_leaves[origin_id] = 0
 266
 267        while True:
 268            if len(open_leaves) == 0:
 269                raise Exception(
 270                    "Something went wrong, the origin and destination nodes are not connected."
 271                )
 272            current_id = min(open_leaves, key=open_leaves.get)
 273            open_leaves.pop(current_id)
 274            if current_id == destination_id:
 275                break
 276            current_distance = distance_matrix[current_id]
 277            for connected_id, connected_distance in graph[current_id].items():
 278                possible_distance = current_distance + connected_distance
 279                if possible_distance < distance_matrix[connected_id]:
 280                    distance_matrix[connected_id] = possible_distance
 281                    predecessor[connected_id] = current_id
 282                    open_leaves[connected_id] = possible_distance

 286            current_id = predecessor[current_id]

 292            "path": output_path,
 293            "length": hard_round(4, distance_matrix[destination_id]),

 457    def get_shortest_path(
 458        self,
 459        origin_node: dict[float | int],
 460        destination_node: dict[float | int],
 461        output_units: str = "km",
 462        algorithm_fn=Graph.dijkstra_makowski,
 463        off_graph_circuity: [float | int] = 1,
 464        node_addition_type: str = "quadrant",
 465        node_addition_circuity: [float | int] = 4,
 466        geograph_units: str = "km",
 467        output_coordinate_path: str = "list_of_lists",
 468        output_path: bool = False,
 469        node_addition_lat_lon_bound: [float | int] = 5,
 470        node_addition_math: str = "euclidean",
 471        **kwargs,
 472    ) -> dict:
 473        """
 474        Function:
 475
 476        - Identify the shortest path between two nodes in a sparse network graph
 477
 478        - Return a dictionary of various path information including:
 479            - `id_path`: A list of node ids in the order they are visited
 480            - `path`: A list of nodes  (list of lat then long) in the order they are visited
 481            - `length`: The length of the path
 482
 483         Required Arguments:
 484
 485        - `origin_node`
 486            - Type: dict of int | float
 487            - What: A dictionary with the keys 'latitude' and 'longitude'
 488        - `destination_node`
 489            - Type: dict of int | float
 490            - What: A dictionary with the keys 'latitude' and 'longitude'
 491
 492        Optional Arguments:
 493
 494        - `output_units`
 495            - Type: str
 496            - What: The units in which to return the length of the path
 497            - Default: 'km'
 498            - Options:
 499                - 'km': Kilometers
 500                - 'm': Meters
 501                - 'mi': Miles
 502                - 'ft': Feet
 503        - `algorithm_fn`
 504            - Type: function | method
 505            - What: The algorithm function to identify the shortest path
 506            - Default: 'Graph.dijkstra_makowski'
 507            - Options:
 508                - 'Graph.dijkstra': A modified dijkstra algorithm that uses a sparse distance matrix to identify the shortest path
 509                - 'Graph.dijkstra_makowski': A modified dijkstra algorithm that uses a sparse distance matrix to identify the shortest path
 510                - Any user defined algorithm that takes the arguments:
 511                    - `graph`: A dictionary of dictionaries where the keys are origin node ids and the values are dictionaries of destination node ids and distances
 512                        - See: https://connor-makowski.github.io/scgraph/scgraph/core.html#GeoGraph
 513                    - `origin`: The id of the origin node from the graph dictionary to start the shortest path from
 514                    - `destination`: The id of the destination node from the graph dictionary to end the shortest path at
 515        - `off_graph_circuity`
 516            - Type: int | float
 517            - What: The circuity factor to apply to any distance calculations between your origin and destination nodes and their connecting nodes in the graph
 518            - Default: 1
 519            - Notes:
 520                - For alogrithmic solving purposes, the node_addition_circuity is applied to the origin and destination nodes when they are added to the graph
 521                - This is only applied after an `optimal solution` using the `node_addition_circuity` has been found when it is then adjusted to equal the `off_graph_circuity`
 522        - `node_addition_type`
 523            - Type: str
 524            - What: The type of node addition to use when adding your origin node to the distance matrix
 525            - Default: 'quadrant'
 526            - Options:
 527                - 'quadrant': Add the closest node in each quadrant (ne, nw, se, sw) to the distance matrix for this node
 528                - 'closest': Add only the closest node to the distance matrix for this node
 529                - 'all': Add all nodes to the distance matrix for this node
 530            - Notes:
 531                - `dijkstra_makowski` will operate substantially faster if the `node_addition_type` is set to 'quadrant' or 'closest'
 532                - `dijkstra` will operate at the similar speeds regardless of the `node_addition_type`
 533                - When using `all`, you should consider using `dijkstra` instead of `dijkstra_makowski` as it will be faster
 534                - The destination node is always added as 'all' regardless of the `node_addition_type` setting
 535                    - This guarantees that any destination node will be connected to any origin node regardless of how or where the origin node is added to the graph
 536                - If the passed graph is not a connected graph (meaning it is comprised of multiple disconnected networks)
 537                    - The entrypoints generated using the `node_addition_type` will determine which disconnected networks will be used to calculate the `optimal route`
 538        - `node_addition_circuity`
 539            - Type: int | float
 540            - What: The circuity factor to apply when adding your origin and destination nodes to the distance matrix
 541            - Default: 4
 542            - Note:
 543                - This defaults to 4 to prevent the algorithm from taking a direct route in direction of the destination over some impassible terrain (EG: a maritime network that goes through land)
 544                - A higher value will push the algorithm to join the network at a closer node to avoid the extra distance from the circuity factor
 545                - This is only relevant if `node_addition_type` is set to 'quadrant' or 'all' as it affects the choice on where to enter the graph network
 546                - This factor is used to calculate the node sequence for the `optimal route`, however the reported `length` of the path will be calculated using the `off_graph_circuity` factor
 547        - `geograph_units`
 548            - Type: str
 549            - What: The units of measurement in the geograph data
 550            - Default: 'km'
 551            - Options:
 552                - 'km': Kilometers
 553                - 'm': Meters
 554                - 'mi': Miles
 555                - 'ft': Feet
 556            - Note: In general, all scgraph provided geographs be in kilometers
 557        - `output_coordinate_path`
 558            - Type: str
 559            - What: The format of the output coordinate path
 560            - Default: 'list_of_lists'
 561            - Options:
 562                - 'list_of_dicts': A list of dictionaries with keys 'latitude' and 'longitude'
 563                - 'list_of_lists': A list of lists with the first value being latitude and the second being longitude
 564                - 'list_of_lists_long_first': A list of lists with the first value being longitude and the second being latitude
 565        - `output_path`
 566            - Type: bool
 567            - What: Whether to output the path as a list of geograph node ids (for debugging and other advanced uses)
 568            - Default: False
 569        - `node_addition_lat_lon_bound`
 570            - Type: int | float
 571            - What: Forms a bounding box around the origin and destination nodes as they are added to graph
 572                - Only points on the current graph inside of this bounding box are considered when updating the distance matrix for the origin or destination nodes
 573            - Default: 5
 574            - Note: If no nodes are found within the bounding box, the bounding box is expanded to 180 degrees in all directions (all nodes are considered)
 575            - Note: This is only used when adding a new node (the specified origin and destination) to the graph
 576        - `node_addition_math`
 577            - Type: str
 578            - What: The math to use when calculating the distance between nodes when determining the closest node (or closest quadrant node) to add to the graph
 579            - Default: 'euclidean'
 580            - Options:
 581                - 'euclidean': Use the euclidean distance between nodes. This is much faster but is not as accurate (especially near the poles)
 582                - 'haversine': Use the haversine distance between nodes. This is slower but is an accurate representation of the surface distance between two points on the earth
 583            - Notes:
 584                - Only used if `node_addition_type` is set to 'quadrant' or 'closest'
 585        - `**kwargs`
 586            - Additional keyword arguments. These are included for forwards and backwards compatibility reasons, but are not currently used.
 587        """
 588        original_graph_length = len(self.graph)
 589        # Add the origin and destination nodes to the graph
 590        origin_id = self.add_node(
 591            node=origin_node,
 592            node_addition_type=node_addition_type,
 593            circuity=node_addition_circuity,
 594            lat_lon_bound=node_addition_lat_lon_bound,
 595            node_addition_math=node_addition_math,
 596        )
 597        destination_id = self.add_node(
 598            node=destination_node,
 599            node_addition_type="all",
 600            circuity=node_addition_circuity,
 601            lat_lon_bound=node_addition_lat_lon_bound,
 602            node_addition_math=node_addition_math,
 603        )
 604
 605        try:
 606            output = algorithm_fn(
 607                graph=self.graph,
 608                origin_id=origin_id,
 609                destination_id=destination_id,
 610            )
 611            output["coordinate_path"] = self.get_coordinate_path(output["path"])
 612            output["length"] = self.adujust_circuity_length(
 613                output=output,
 614                node_addition_circuity=node_addition_circuity,
 615                off_graph_circuity=off_graph_circuity,
 616            )
 617            output["length"] = distance_converter(
 618                output["length"],
 619                input_units=geograph_units,
 620                output_units=output_units,
 621            )
 622            if output_coordinate_path == "list_of_dicts":
 623                output["coordinate_path"] = [
 624                    {"latitude": i[0], "longitude": i[1]}
 625                    for i in output["coordinate_path"]
 626                ]
 627            elif output_coordinate_path == "list_of_lists_long_first":
 628                output["coordinate_path"] = [
 629                    [i[1], i[0]] for i in output["coordinate_path"]
 630                ]
 631                output["long_first"] = True
 632            if not output_path:
 633                del output["path"]
 634            while len(self.graph) > original_graph_length:
 635                self.remove_appended_node()

 637        except Exception as e:
 638            while len(self.graph) > original_graph_length:

 642    def adujust_circuity_length(
 643        self,
 644        output: dict,
 645        node_addition_circuity: [float | int],
 646        off_graph_circuity: [float | int],
 647    ) -> [float | int]:
 648        """
 649        Function:
 650
 651        - Adjust the length of the path to account for the circuity factors applied to the origin and destination nodes
 652
 653        Required Arguments:
 654
 655        - `output`
 656            - Type: dict
 657            - What: The output from the algorithm function
 658        - `node_addition_circuity`
 659            - Type: int | float
 660            - What: The circuity factor that was applied when adding your origin and destination nodes to the distance matrix
 661        - `off_graph_circuity`
 662            - Type: int | float
 663            - What: The circuity factor to apply to any distance calculations between your origin and destination nodes and their connecting nodes in the graph
 664        """
 665        coordinate_path = output["coordinate_path"]
 666        # If the path does not enter the graph, undo the node_addition_circuity and apply the off_graph_circuity
 667        if len(output["coordinate_path"]) == 2:
 668            return (
 669                output["length"] / node_addition_circuity
 670            ) * off_graph_circuity
 671        else:
 672            direct_off_graph_length = haversine(
 673                coordinate_path[0], coordinate_path[1], circuity=1
 674            ) + haversine(coordinate_path[-2], coordinate_path[-1], circuity=1)
 675            return round(
 676                output["length"]
 677                + direct_off_graph_length
 678                * (off_graph_circuity - node_addition_circuity),
 679                4,
 680            )

 723    def get_node_distances(
 724        self,
 725        node: list,
 726        circuity: [float | int],
 727        node_addition_type: str,
 728        node_addition_math: str,
 729        lat_lon_bound: [float | int],
 730    ) -> dict[float | int]:
 731        """
 732        Function:
 733
 734        - Get the distances between a node and all other nodes in the graph
 735        - This is used to determine the closest node to add to the graph when adding a new node
 736
 737        Required Arguments:
 738
 739        - `node`
 740            - Type: list
 741            - What: A list of the latitude and longitude of the node
 742            - EG: [latitude, longitude] -> [31.23, 121.47]
 743        - `circuity`
 744            - Type: int | float
 745            - What: The circuity to apply to any distance calculations
 746            - Note: This defaults to 4 to prevent the algorithm from taking a direct route in direction of the destination over some impassible terrain (EG: a maritime network that goes through land)
 747        - `node_addition_type`
 748            - Type: str
 749            - What: The type of node addition to use
 750            - Options:
 751                - 'quadrant': Add the closest node in each quadrant (ne, nw, se, sw) to the distance matrix for this node
 752                - 'closest': Add only the closest node to the distance matrix for this node
 753                - 'all': Add all nodes to the distance matrix for this node
 754            - Notes:
 755                - `dijkstra_makowski` will operate substantially faster if the `node_addition_type` is set to 'quadrant' or 'closest'
 756                - `dijkstra` will operate at the similar speeds regardless of the `node_addition_type`
 757                - When using `all`, you should consider using `dijkstra` instead of `dijkstra_makowski` as it will be faster
 758        - `node_addition_math`
 759            - Type: str
 760            - What: The math to use when calculating the distance between nodes when determining the closest node (or closest quadrant node) to add to the graph
 761            - Default: 'euclidean'
 762            - Options:
 763                - 'euclidean': Use the euclidean distance between nodes. This is much faster but is not accurate (especially near the poles)
 764                - 'haversine': Use the haversine distance between nodes. This is slower but is an accurate representation of the surface distance between two points on the earth
 765            - Notes:
 766                - Once the closest node (or closest quadrant node) is determined, the haversine distance (with circuity) is used to calculate the distance between the nodes when adding it to the graph.
 767        - `lat_lon_bound`
 768            - Type: int | float
 769            - What: Forms a bounding box around the node that is to be added to graph. Only selects graph nodes to consider joining that are within this bounding box.
 770        """
 771        assert node_addition_type in [
 772            "quadrant",
 773            "all",
 774            "closest",
 775        ], f"Invalid node addition type provided ({node_addition_type}), valid options are: ['quadrant', 'all', 'closest']"
 776        assert node_addition_math in [
 777            "euclidean",
 778            "haversine",
 779        ], f"Invalid node addition math provided ({node_addition_math}), valid options are: ['euclidean', 'haversine']"
 780        # Get only bounded nodes
 781        nodes = {
 782            node_idx: node_i
 783            for node_idx, node_i in enumerate(self.nodes)
 784            if abs(node_i[0] - node[0]) < lat_lon_bound
 785            and abs(node_i[1] - node[1]) < lat_lon_bound
 786        }
 787        if len(nodes) == 0:
 788            # Default to all if the lat_lon_bound fails to find any nodes
 789            return self.get_node_distances(
 790                node=node,
 791                circuity=circuity,
 792                lat_lon_bound=180,
 793                node_addition_type=node_addition_type,
 794                node_addition_math=node_addition_math,
 795            )
 796        if node_addition_type == "all":
 797            return {
 798                node_idx: round(haversine(node, node_i, circuity=circuity), 4)
 799                for node_idx, node_i in nodes.items()
 800            }
 801        if node_addition_math == "haversine":
 802            dist_fn = lambda x: round(haversine(node, x, circuity=circuity), 4)
 803        else:
 804            dist_fn = lambda x: round(
 805                ((node[0] - x[0]) ** 2 + (node[1] - x[1]) ** 2) ** 0.5, 4
 806            )
 807        if node_addition_type == "closest":
 808            quadrant_fn = lambda x, y: "all"
 809        else:
 810            quadrant_fn = lambda x, y: ("n" if x[0] - y[0] > 0 else "s") + (
 811                "e" if x[1] - y[1] > 0 else "w"
 812            )
 813        min_diffs = {}
 814        min_diffs_idx = {}
 815        for node_idx, node_i in nodes.items():
 816            quadrant = quadrant_fn(node_i, node)
 817            dist = dist_fn(node_i)
 818            if dist < min_diffs.get(quadrant, 999999999):
 819                min_diffs[quadrant] = dist
 820                min_diffs_idx[quadrant] = node_idx
 821        return {
 822            node_idx: round(
 823                haversine(node, self.nodes[node_idx], circuity=circuity), 4
 824            )
 825            for node_idx in min_diffs_idx.values()
 826        }

232                - Open leaves are nodes that have not been visited yet but are adjacent to other visited nodes
233            - This can dramatically reduce the memory and compute requirements of the algorithm
234            - For particularly sparse graphs, this algorithm runs close to O(n log n) time
235                - Where n is the number of nodes in the graph
236            - For dense graphs, this algorithm runs closer to O(n^2) time (similar to the standard Dijkstra algorithm)
237                - Where n is the number of nodes in the graph

250        - `destination_id`

253
254        Optional Arguments:
255
256        - None
257        """

259            graph=graph, origin_id=origin_id, destination_id=destination_id
260        )
261        distance_matrix = [float("inf") for i in graph]
262        open_leaves = {}
263        predecessor = [None for i in graph]
264
265        distance_matrix[origin_id] = 0
266        open_leaves[origin_id] = 0
267
268        while True:
269            if len(open_leaves) == 0:
270                raise Exception(
271                    "Something went wrong, the origin and destination nodes are not connected."
272                )
273            current_id = min(open_leaves, key=open_leaves.get)
274            open_leaves.pop(current_id)
275            if current_id == destination_id:
276                break
277            current_distance = distance_matrix[current_id]
278            for connected_id, connected_distance in graph[current_id].items():
279                possible_distance = current_distance + connected_distance
280                if possible_distance < distance_matrix[connected_id]:
281                    distance_matrix[connected_id] = possible_distance
282                    predecessor[connected_id] = current_id
283                    open_leaves[connected_id] = possible_distance
284
285        output_path = [current_id]
286        while predecessor[current_id] is not None:
287            current_id = predecessor[current_id]
288            output_path.append(current_id)
289
290        output_path.reverse()
291

294            "length": hard_round(4, distance_matrix[destination_id]),
