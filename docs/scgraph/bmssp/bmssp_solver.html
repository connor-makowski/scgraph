  9def cnt_reachable_nodes(root: int, forest: dict[int, set[int]]) -> int:
 10    """
 11    Function:
 12
 13    - Return the number of nodes reachable from root in the directed forest defined by forest.
 14
 15    Required Arguments:
 16
 17    - `root`
 18        - Type: int
 19        - What: The starting node for the DFS traversal.
 20    - `forest`
 21        - Type: dict[int, set[int]]
 22        - What: Adjacency list representing the directed forest.
 23    """
 24    seen = set()
 25    stack = [root]
 26    cnt = 0
 27    while stack:
 28        x = stack.pop()
 29        if x in seen:
 30            continue
 31        seen.add(x)
 32        cnt += 1
 33        stack.extend(forest[x])
 34    return cnt

 42        Function:
 43
 44        - Initialize the BMSSP solver with a graph represented as an adjacency list.
 45
 46        Required Arguments:
 47
 48        - graph:
 49            - Type: list[dict[int, int | float]]
 50            - Description: The graph is represented as an adjacency list, where each node points to a dictionary of its neighbors and their edge weights.
 51        - origin_ids:
 52            - Type: set[int] | int
 53            - What: The IDs of the starting nodes for the BMSSP algorithm.
 54            - Note: Can be a single integer or a set of integers.
 55
 56        Optional Arguments:
 57
 58        - None
 59        """
 60        if isinstance(origin_ids, int):
 61            origin_ids = {origin_ids}
 62        self.graph = graph
 63        graph_len = len(graph)
 64        self.distance_matrix = [inf] * graph_len
 65        # Addition: Initialize Predecessor array for path reconstruction
 66        self.predecessor = [-1] * graph_len
 67        for origin_id in origin_ids:
 68            self.distance_matrix[origin_id] = 0
 69
 70        if graph_len <= 2:
 71            raise ValueError("Your provided graph must have more than 2 nodes")
 72
 73        # Practical choices (k and t) based on n
 74        self.pivot_relaxation_steps = max(
 75            2, int(log(graph_len) ** (1 / 3.0))
 76        )  # k
 77        # Modification: Change int to ceil
 78        self.target_tree_depth = max(2, ceil(log(graph_len) ** (2 / 3.0)))  # t
 79
 80        # Compute max_tree_depth based on k and t
 81        self.max_tree_depth = int(ceil(log(graph_len) / self.target_tree_depth))
 82
 83        # Run the solver algorithm
 84        upper_bound, frontier = self.recursive_bmssp(
 85            self.max_tree_depth, inf, origin_ids

 88    def find_pivots(
 89        self, upper_bound: int | float, frontier: set[int]
 90    ) -> tuple[set[int], set[int]]:
 91        """
 92        Function:
 93
 94        - Finds pivot sets pivots and temp_frontier according to Algorithm 1.
 95
 96        Required Arguments:
 97
 98        - upper_bound:
 99            - Type: int | float
100            - What: The upper bound threshold (B)
101        - frontier:
102            - Type: set[int]
103            - What: Set of vertices (S)
104
105        Optional Arguments:
106
107        - None
108
109        Returns:
110
111        - pivots:
112            - Type: Set[int]
113            - What: Set of pivot vertices
114        - frontier:
115            - Type: Set[int]
116            - What: Return a new frontier set of vertices within the upper_bound
117        """
118        temp_frontier = set(frontier)
119        prev_frontier = set(frontier)
120
121        # Multi-step limited relaxation from current frontier
122        for _ in range(self.pivot_relaxation_steps):
123            curr_frontier = set()
124            for prev_frontier_idx in prev_frontier:
125                prev_frontier_distance = self.distance_matrix[prev_frontier_idx]
126                for connection_idx, connection_distance in self.graph[
127                    prev_frontier_idx
128                ].items():
129                    new_distance = prev_frontier_distance + connection_distance
130                    # Important: Allow equality on relaxations
131                    if new_distance <= self.distance_matrix[connection_idx]:
132                        # Addition: Add predecessor tracking
133                        if new_distance < self.distance_matrix[connection_idx]:
134                            self.predecessor[connection_idx] = prev_frontier_idx
135                            self.distance_matrix[connection_idx] = new_distance
136                        if new_distance < upper_bound:
137                            curr_frontier.add(connection_idx)
138            temp_frontier.update(curr_frontier)
139            # If the search balloons, take the current frontier as pivots
140            if len(temp_frontier) > self.pivot_relaxation_steps * len(frontier):
141                pivots = set(frontier)
142                return pivots, temp_frontier
143            prev_frontier = curr_frontier
144
145        # Build tight-edge forest F on temp_frontier: edges (u -> v) with db[u] + w == db[v]
146        forest = {i: set() for i in temp_frontier}
147        indegree = {i: 0 for i in temp_frontier}
148        for frontier_idx in temp_frontier:
149            frontier_distance = self.distance_matrix[frontier_idx]
150            for connection_idx, connection_distance in self.graph[
151                frontier_idx
152            ].items():
153                if (
154                    connection_idx in temp_frontier
155                    and (frontier_distance + connection_distance)
156                    == self.distance_matrix[connection_idx]
157                ):
158                    # direction is frontier_idx -> connection_idx (parent to child)
159                    forest[frontier_idx].add(connection_idx)
160                    indegree[connection_idx] += 1
161
162        pivots = set()
163        for frontier_idx in frontier:
164            if indegree.get(frontier_idx, 0) == 0:
165                size = cnt_reachable_nodes(frontier_idx, forest=forest)
166                if size >= self.pivot_relaxation_steps:
167                    pivots.add(frontier_idx)
168
169        return pivots, temp_frontier

171    def base_case(
172        self, upper_bound: int | float, frontier: set[int]
173    ) -> tuple[int | float, set[int]]:
174        """
175        Function:
176
177        - Implements Algorithm 2: Base Case of BMSSP
178
179        Required Arguments:
180        - upper_bound:
181            - Type: int | float
182        - frontier:
183            - Type: set
184            - What: Set with a single vertex x (complete)
185
186        Returns:
187        - new_upper_bound:
188            - Type: int | float
189            - What: The new upper bound for the search
190        - new_frontier:
191            - Type: set[int]
192            - What: Set of vertices v such that distance_matrix[v] < new_upper_bound
193        """
194        assert len(frontier) == 1, "Frontier must be a singleton set"
195        first_frontier = next(iter(frontier))
196
197        new_frontier = set()
198        heap = []
199        heappush(heap, (self.distance_matrix[first_frontier], first_frontier))
200        # Grow until we exceed pivot_relaxation_steps (practical limit), as in Algorithm 2
201        while heap and len(new_frontier) < self.pivot_relaxation_steps + 1:
202            frontier_distance, frontier_idx = heappop(heap)
203            # Addition: Add check to ensure that we do not get caught in a relaxation loop
204            if frontier_idx in new_frontier:
205                continue
206            new_frontier.add(frontier_idx)
207            for connection_idx, connection_distance in self.graph[
208                frontier_idx
209            ].items():
210                new_distance = frontier_distance + connection_distance
211                if (
212                    new_distance <= self.distance_matrix[connection_idx]
213                    and new_distance < upper_bound
214                ):
215                    # Addition: Add predecessor tracking
216                    if new_distance < self.distance_matrix[connection_idx]:
217                        self.predecessor[connection_idx] = frontier_idx
218                        self.distance_matrix[connection_idx] = new_distance
219                    heappush(heap, (new_distance, connection_idx))
220
221        # If we exceeded k, trim by new boundary B' = max db over visited and return new_frontier = {db < B'}
222        if len(new_frontier) > self.pivot_relaxation_steps:
223            new_upper_bound = max(self.distance_matrix[i] for i in new_frontier)
224            new_frontier = {
225                i
226                for i in new_frontier
227                if self.distance_matrix[i] < new_upper_bound
228            }
229            return new_upper_bound, new_frontier
230        else:
231            # Success for this base case: return current upper_bound unchanged and the completed set
232            return upper_bound, new_frontier

234    def recursive_bmssp(
235        self, recursion_depth: int, upper_bound: int | float, frontier: set[int]
236    ) -> tuple[int | float, set[int]]:
237        """
238        Function:
239
240        - Implements Algorithm 3: Bounded Multi-Source Shortest Path (BMSSP)
241
242        Required Arguments:
243
244        - recursion_depth:
245            - Type: int
246            - What: The depth of the recursion
247        - upper_bound:
248            - Type: float
249            - What: The upper bound for the search
250        - frontier:
251            - Type: set[int]
252            - What: The set of vertices to explore
253
254        Returns:
255
256        - new_upper_bound:
257            - Type: int | float
258            - What: The new upper bound for the search
259        - new_frontier:
260            - Type: set[int]
261            - What: Set of vertices v such that distance_matrix[v] < new_upper_bound
262        """
263        # Base case
264        if recursion_depth == 0:
265            return self.base_case(upper_bound, frontier)
266
267        # Step 4: Find pivots and temporary frontier
268        pivots, temp_frontier = self.find_pivots(upper_bound, frontier)
269
270        # Step 5–6: initialize data_struct with pivots
271        # subset_size = 2^((l-1) * t)
272        subset_size = 2 ** ((recursion_depth - 1) * self.target_tree_depth)
273        data_struct = BmsspDataStructure(
274            subset_size=subset_size, upper_bound=upper_bound
275        )
276
277        for p in pivots:
278            data_struct.insert_key_value(p, self.distance_matrix[p])
279
280        # Track new_frontier and B' according to Algorithm 3
281        new_frontier = set()
282        # Store the min_pivot_distance for use if the frontier is empty and we break early
283        min_pivot_distance = min(
284            (self.distance_matrix[p] for p in pivots), default=upper_bound
285        )
286
287        # Work budget that scales with level: k^(2*l*t)
288        # k = self.pivot_relaxation_steps
289        # t = self.target_tree_depth
290        work_budget = self.pivot_relaxation_steps ** (
291            2 * recursion_depth * self.target_tree_depth
292        )
293
294        # Main loop
295        while len(new_frontier) < work_budget and not data_struct.is_empty():
296            # Step 10: Pull from data_struct: get data_struct_frontier_temp and upper_bound_i
297            data_struct_frontier_bound_temp, data_struct_frontier_temp = (
298                data_struct.pull()
299            )
300
301            # Step 11: Recurse on (l-1, data_struct_frontier_bound_temp, data_struct_frontier_temp)
302            min_pivot_distance, new_frontier_temp = self.recursive_bmssp(
303                recursion_depth - 1,
304                data_struct_frontier_bound_temp,
305                data_struct_frontier_temp,
306            )
307
308            # Track results
309            new_frontier.update(new_frontier_temp)
310
311            # Step 13: Initialize intermediate_frontier to batch-prepend
312            intermediate_frontier = set()
313
314            # Step 14–20: relax edges from new_frontier_temp and enqueue into D or intermediate_frontier per their interval
315            for new_frontier_idx in new_frontier_temp:
316                new_frontier_distance = self.distance_matrix[new_frontier_idx]
317                for connection_idx, connection_distance in self.graph[
318                    new_frontier_idx
319                ].items():
320                    # Addition: Avoid self-loops
321                    if connection_idx == new_frontier_idx:
322                        continue
323                    new_distance = new_frontier_distance + connection_distance
324                    if new_distance <= self.distance_matrix[connection_idx]:
325                        # Addition: Add predecessor tracking
326                        if new_distance < self.distance_matrix[connection_idx]:
327                            self.predecessor[connection_idx] = new_frontier_idx
328                            self.distance_matrix[connection_idx] = new_distance
329                        # Insert based on which interval the new distance falls into
330                        if (
331                            data_struct_frontier_bound_temp
332                            <= new_distance
333                            < upper_bound
334                        ):
335                            data_struct.insert_key_value(
336                                connection_idx, new_distance
337                            )
338                        elif (
339                            min_pivot_distance
340                            <= new_distance
341                            < data_struct_frontier_bound_temp
342                        ):
343                            intermediate_frontier.add(
344                                (connection_idx, new_distance)
345                            )
346
347            # Step 21: Batch prepend intermediate_frontier plus filtered data_struct_frontier_temp in min_pivot_distance, data_struct_frontier_bound_temp)
348            data_struct_frontier_temp_filtered = {
349                (x, self.distance_matrix[x])
350                for x in data_struct_frontier_temp
351                if min_pivot_distance
352                <= self.distance_matrix[x]
353                < data_struct_frontier_bound_temp
354            }
355
356            data_struct.batch_prepend(
357                intermediate_frontier | data_struct_frontier_temp_filtered
358            )
359
360        # Step 22: Final return
361        new_bound = min(min_pivot_distance, upper_bound)
362        return new_bound, new_frontier | {
363            v for v in temp_frontier if self.distance_matrix[v] < new_bound
364        }
