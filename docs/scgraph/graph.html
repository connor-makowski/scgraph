 12        """
 13        Function:
 14
 15        - Calculate the shortest path tree of a graph using Dijkstra's algorithm
 16
 17        Required Arguments:
 18
 19        - `origin_id`
 20            - Type: int | set[int]
 21            - What: The id(s) of the node(s) from which to calculate the shortest path tree
 22
 23        Returns:
 24
 25        - A dictionary with the following keys:
 26            - `origin_id`: The id of the node from which the shortest path tree was calculated
 27            - `predecessors`: A list of node ids referring to the predecessor of each node given the shortest path tree
 28                - Note: For disconnected graphs, nodes that are not connected to the origin node will have a predecessor of None
 29            - `distance_matrix`: A list of distances from the origin node to each node in the graph
 30                - Note: For disconnected graphs, nodes that are not connected to the origin node will have a distance of float("inf")
 31
 32        """
 33        # Input Validation
 34        self.__input_check__(origin_id=origin_id, destination_id=0)
 35        origin_ids = {origin_id} if isinstance(origin_id, int) else origin_id
 36
 37        # Variable Initialization
 38        distance_matrix = [float("inf")] * len(self.graph)
 39        open_leaves = []
 40        predecessor = [-1] * len(self.graph)
 41
 42        for oid in origin_ids:
 43            distance_matrix[oid] = 0
 44            heappush(open_leaves, (0, oid))
 45
 46        while open_leaves:
 47            current_distance, current_id = heappop(open_leaves)
 48            for connected_id, connected_distance in self.graph[
 49                current_id
 50            ].items():
 51                possible_distance = current_distance + connected_distance
 52                if possible_distance < distance_matrix[connected_id]:
 53                    distance_matrix[connected_id] = possible_distance
 54                    predecessor[connected_id] = current_id
 55                    heappush(open_leaves, (possible_distance, connected_id))
 56
 57        return {
 58            "origin_id": origin_id,
 59            "predecessors": predecessor,
 60            "distance_matrix": distance_matrix,
 61        }

 63    def get_tree_path(
 64        self,
 65        origin_id: int,
 66        destination_id: int,
 67        tree_data: dict,
 68        length_only: bool = False,
 69    ) -> dict:
 70        """
 71        Function:
 72
 73        - Get the path from the origin node to the destination node using the provided tree_data
 74        - Return a list of node ids in the order they are visited as well as the length of the path
 75        - If the destination node is not reachable from the origin node, return None
 76
 77        Required Arguments:
 78
 79        - `origin_id`
 80            - Type: int
 81            - What: The id of the origin node from the graph dictionary to start the shortest path from
 82            - Note: Since multiple origins are possible, if the origin_id is not a predecessor node of the destination node, the closest origin will be used.
 83        - `destination_id`
 84            - Type: int
 85            - What: The id of the destination node from the graph dictionary to end the shortest path at
 86        - `tree_data`
 87            - Type: dict
 88            - What: The output from a tree function
 89
 90        Optional Arguments:
 91
 92        - `length_only`
 93            - Type: bool
 94            - What: If True, only returns the length of the path
 95            - Default: False
 96
 97        Returns:
 98
 99        - A dictionary with the following keys:
100            - `path`: A list of node ids in the order they are visited from the origin node to the destination node
101            - `length`: The length of the path from the origin node to the destination node
102        """
103        if tree_data["origin_id"] != origin_id:
104            raise Exception(
105                "The origin node must be the same as the spanning node for this function to work."
106            )
107        destination_distance = tree_data["distance_matrix"][destination_id]
108        if destination_distance == float("inf"):
109            raise Exception(
110                "Something went wrong, the origin and destination nodes are not connected."
111            )
112        if length_only:
113            return {"length": destination_distance}
114        current_id = destination_id
115        current_path = [destination_id]
116        while current_id != origin_id and current_id != -1:
117            current_id = tree_data["predecessors"][current_id]
118            current_path.append(current_id)
119        current_path.reverse()
120        return {
121            "path": current_path,
122            "length": destination_distance,
123        }

128        self,
129        origin_id: int | set[int],
130        destination_id: int,
131    ) -> dict:
132        """
133        Function:
134
135        - Identify the shortest path between two nodes in a sparse network graph using a modified Dijkstra algorithm
136
137        - Return a dictionary of various path information including:
138            - `path`: A list of node ids in the order they are visited

140
141        Required Arguments:
142
143        - `origin_id`
144            - Type: int | set[int]
145            - What: The id(s) of the origin node(s) from the graph dictionary to start the shortest path from
146        - `destination_id`
147            - Type: int
148            - What: The id of the destination node from the graph dictionary to end the shortest path at
149
150        Optional Arguments:
151
152        - None
153        """
154        # Input Validation
155        self.__input_check__(origin_id=origin_id, destination_id=destination_id)
156        origin_ids = {origin_id} if isinstance(origin_id, int) else origin_id
157        # Variable Initialization
158        distance_matrix = [float("inf")] * len(self.graph)
159        predecessor = [-1] * len(self.graph)
160        open_leaves = []
161
162        for oid in origin_ids:
163            distance_matrix[oid] = 0
164            heappush(open_leaves, (0, oid))
165
166        while open_leaves:
167            current_distance, current_id = heappop(open_leaves)
168            if current_id == destination_id:
169                break
170            # Technically, the next line is not necessary but can help with performance
171            if current_distance == distance_matrix[current_id]:
172                for connected_id, connected_distance in self.graph[

174                ].items():
175                    possible_distance = current_distance + connected_distance
176                    if possible_distance < distance_matrix[connected_id]:
177                        distance_matrix[connected_id] = possible_distance
178                        predecessor[connected_id] = current_id
179                        heappush(open_leaves, (possible_distance, connected_id))
180        if current_id != destination_id:
181            raise Exception(
182                "Something went wrong, the origin and destination nodes are not connected."
183            )
184
185        return {
186            "path": self.__reconstruct_path__(destination_id, predecessor),
187            "length": distance_matrix[destination_id],
188        }

190    def dijkstra_buckets(
191        self,
192        origin_id: int | set[int],
193        destination_id: int,
194        max_edge_weight: int | float | None = None,
195    ) -> dict:
196        """
197        Function:
198
199        - Identify the shortest path between two nodes in a sparse network graph using Dijkstra's algorithm with buckets (Dial's algorithm)
200        - This is particularly efficient for graphs where most edge weights are >= 1 and the maximum edge weight is small
201        - This implementation safely supports non-integer weights
202        - Return a dictionary of various path information including:
203            - `path`: A list of node ids in the order they are visited
204            - `length`: The length of the path from the origin node to the destination node
205
206        Required Arguments:
207
208        - `origin_id`
209            - Type: int | set[int]
210            - What: The id(s) of the origin node(s) from the graph dictionary to start the shortest path from
211        - `destination_id`
212            - Type: int
213            - What: The id of the destination node from the graph dictionary to end the shortest path at
214
215        Optional Arguments:
216
217        - `max_edge_weight`
218            - Type: int | float | None
219            - What: The maximum edge weight in the graph. If None, it will be calculated.
220            - Default: None
221        """
222        # Input Validation
223        self.__input_check__(origin_id=origin_id, destination_id=destination_id)
224        origin_ids = {origin_id} if isinstance(origin_id, int) else origin_id
225
226        if max_edge_weight is None:
227            max_edge_weight = 0
228            for node_edges in self.graph:
229                if node_edges:
230                    m = max(node_edges.values())
231                    if m > max_edge_weight:
232                        max_edge_weight = m
233        max_edge_weight = math.ceil(max_edge_weight)
234
235        # Variable Initialization
236        distance_matrix = [float("inf")] * len(self.graph)
237        predecessor = [-1] * len(self.graph)
238        num_buckets = max_edge_weight + 1
239        buckets = [[] for _ in range(num_buckets)]
240
241        for oid in origin_ids:
242            distance_matrix[oid] = 0
243            buckets[0].append(oid)
244
245        current_dist = 0
246        nodes_in_buckets = len(origin_ids)
247
248        while nodes_in_buckets > 0:
249            bucket_idx = current_dist % num_buckets
250            while not buckets[bucket_idx]:
251                current_dist += 1
252                bucket_idx = current_dist % num_buckets
253                if nodes_in_buckets == 0:
254                    break
255                # If we've already found a path shorter than the current bucket minimum, we can exit
256                if distance_matrix[destination_id] < current_dist:
257                    break
258
259            if (
260                nodes_in_buckets == 0
261                or distance_matrix[destination_id] < current_dist
262            ):
263                break
264
265            current_id = buckets[bucket_idx].pop()
266            nodes_in_buckets -= 1
267
268            # Skip if we found a better path already (lazy removal)
269            if distance_matrix[current_id] < current_dist:
270                continue
271
272            # Note: We do not break immediately if current_id == destination_id
273            # because weights < 1 might allow a shorter path to be found within the same bucket.
274            # The loop terminates when current_dist > distance_matrix[destination_id].
275
276            for connected_id, connected_distance in self.graph[
277                current_id
278            ].items():
279                possible_distance = (
280                    distance_matrix[current_id] + connected_distance
281                )
282                if possible_distance < distance_matrix[connected_id]:
283                    distance_matrix[connected_id] = possible_distance
284                    predecessor[connected_id] = current_id
285                    buckets[int(possible_distance) % num_buckets].append(
286                        connected_id
287                    )
288                    nodes_in_buckets += 1
289
290        if distance_matrix[destination_id] == float("inf"):
291            raise Exception(
292                "Something went wrong, the origin and destination nodes are not connected."
293            )
294
295        return {
296            "path": self.__reconstruct_path__(destination_id, predecessor),
297            "length": distance_matrix[destination_id],
298        }

300    def dijkstra_negative(
301        self,
302        origin_id: int | set[int],
303        destination_id: int,
304        cycle_check_iterations: int | None = None,
305    ) -> dict:
306        """
307        Function:
308
309        - Identify the shortest path between two nodes in a sparse network graph using a modified Dijkstra algorithm that catches negative cycles
310            - Negative cycles raise an exception if they are detected
311        - Note: This algorithm is guaranteed to find the shortest path or raise an exception if a negative cycle is detected
312        - Note: This algorithm requires computing the entire shortest path tree of the graph and is therefore not able to be terminated early
313            - For non negative weighted graphs, it is recommended to use the `dijkstra` algorithm instead
314        - Note: This should be fairly performant in general, however it does have a higher worst-case time complexity than Bellman-Ford
315
316        - Return a dictionary of various path information including:
317            - `id_path`: A list of node ids in the order they are visited
318            - `path`: A list of node dictionaries (lat + long) in the order they are visited
319
320        Required Arguments:
321
322        - `origin_id`
323            - Type: int | set[int]
324            - What: The id(s) of the origin node(s) from the graph dictionary to start the shortest path from
325        - `destination_id`
326            - Type: int
327            - What: The id of the destination node from the graph dictionary to end the shortest path at
328
329        Optional Arguments:
330
331        - `cycle_check_iterations`
332            - Type: int | None
333            - What: The number of iterations to loop over before checking for negative cycles
334            - Default: None (Then set to the number of nodes in the graph)
335        """
336        # Input Validation
337        self.__input_check__(origin_id=origin_id, destination_id=destination_id)
338        origin_ids = {origin_id} if isinstance(origin_id, int) else origin_id
339
340        # Variable Initialization
341        distance_matrix = [float("inf")] * len(self.graph)
342        predecessor = [-1] * len(self.graph)
343        open_leaves = []
344
345        for oid in origin_ids:
346            distance_matrix[oid] = 0
347            heappush(open_leaves, (0, oid))
348
349        # Cycle iteration Variables
350        cycle_iteration = 0
351        if cycle_check_iterations is None:
352            cycle_check_iterations = len(self.graph)
353
354        while open_leaves:
355            current_distance, current_id = heappop(open_leaves)
356            if current_distance == distance_matrix[current_id]:
357                # Increment the cycle iteration counter and check for negative cycles if the iteration limit is reached
358                cycle_iteration += 1
359                if cycle_iteration >= cycle_check_iterations:
360                    cycle_iteration = 0  # Reset the cycle iteration counter
361                    self.__cycle_check__(
362                        predecessor_matrix=predecessor, node_id=current_id
363                    )
364                for connected_id, connected_distance in self.graph[
365                    current_id
366                ].items():
367                    possible_distance = current_distance + connected_distance
368                    if possible_distance < distance_matrix[connected_id]:
369                        distance_matrix[connected_id] = possible_distance
370                        predecessor[connected_id] = current_id
371                        heappush(open_leaves, (possible_distance, connected_id))
372
373        if distance_matrix[destination_id] == float("inf"):
374            raise Exception(
375                "Something went wrong, the origin and destination nodes are not connected."
376            )
377
378        return {
379            "path": self.__reconstruct_path__(destination_id, predecessor),
380            "length": distance_matrix[destination_id],
381        }

383    def a_star(
384        self,
385        origin_id: int | set[int],
386        destination_id: int,
387        heuristic_fn: callable = None,
388    ) -> dict:
389        """
390        Function:
391
392        - Identify the shortest path between two nodes in a sparse network graph using an A* extension of Makowski's modified Dijkstra algorithm
393        - Return a dictionary of various path information including:
394            - `id_path`: A list of node ids in the order they are visited
395            - `path`: A list of node dictionaries (lat + long) in the order they are visited
396
397        Required Arguments:
398
399        - `origin_id`
400            - Type: int | set[int]
401            - What: The id of the origin node from the graph dictionary to start the shortest path from
402        - `destination_id`
403            - Type: int
404            - What: The id of the destination node from the graph dictionary to end the shortest path at
405        - `heuristic_fn`
406            - Type: function
407            - What: A heuristic function that takes two node ids and returns an estimated distance between them
408            - Note: If None, returns the shortest path using Dijkstra's algorithm
409            - Default: None
410
411        Optional Arguments:
412
413        - None
414        """
415        if heuristic_fn is None:
416            return self.dijkstra(
417                origin_id=origin_id,
418                destination_id=destination_id,
419            )
420        # Input Validation
421        self.__input_check__(origin_id=origin_id, destination_id=destination_id)
422        origin_ids = {origin_id} if isinstance(origin_id, int) else origin_id
423
424        # Variable Initialization
425        distance_matrix = [float("inf")] * len(self.graph)
426        # Using a visited matrix does add a tad bit of overhead but avoids revisiting nodes
427        # and does not require anything extra to be stored in the heap
428        visited = [0] * len(self.graph)
429        open_leaves = []
430        predecessor = [-1] * len(self.graph)
431
432        for oid in origin_ids:
433            distance_matrix[oid] = 0
434            heappush(open_leaves, (0, oid))
435
436        while open_leaves:
437            current_id = heappop(open_leaves)[1]
438            if current_id == destination_id:
439                break
440            if visited[current_id] == 1:
441                continue
442            visited[current_id] = 1
443            current_distance = distance_matrix[current_id]
444            for connected_id, connected_distance in self.graph[
445                current_id
446            ].items():
447                possible_distance = current_distance + connected_distance
448                if possible_distance < distance_matrix[connected_id]:
449                    distance_matrix[connected_id] = possible_distance
450                    predecessor[connected_id] = current_id
451                    heappush(
452                        open_leaves,
453                        (
454                            possible_distance
455                            + heuristic_fn(connected_id, destination_id),
456                            connected_id,
457                        ),
458                    )
459        if current_id != destination_id:
460            raise Exception(
461                "Something went wrong, the origin and destination nodes are not connected."
462            )
463
464        return {
465            "path": self.__reconstruct_path__(destination_id, predecessor),
466            "length": distance_matrix[destination_id],
467        }

470        self,
471        origin_id: int | set[int],
472        destination_id: int,
473    ) -> dict:
474        """
475        Function:
476
477        - Identify the shortest path between two nodes in a sparse network graph using the Bellman-Ford algorithm
478        - Return a dictionary of various path information including:
479            - `id_path`: A list of node ids in the order they are visited
480            - `path`: A list of node dictionaries (lat + long) in the order they are visited
481
482        Required Arguments:
483
484        - `origin_id`
485            - Type: int | set[int]
486            - What: The id(s) of the origin node(s) from the graph dictionary to start the shortest path from
487        - `destination_id`
488            - Type: int
489            - What: The id of the destination node from the graph dictionary to end the shortest path at
490
491        Optional Arguments:
492
493        - None
494        """
495        # Input Validation
496        self.__input_check__(origin_id=origin_id, destination_id=destination_id)

498        # Variable Initialization
499        distance_matrix = [float("inf")] * len(self.graph)
500        predecessor = [-1] * len(self.graph)
501
502        for oid in origin_ids:
503            distance_matrix[oid] = 0
504
505        len_graph = len(self.graph)
506        for i in range(len_graph):
507            for current_id in range(len(self.graph)):
508                current_distance = distance_matrix[current_id]
509                for connected_id, connected_distance in self.graph[
510                    current_id
511                ].items():
512                    possible_distance = current_distance + connected_distance
513                    if possible_distance < distance_matrix[connected_id]:
514                        distance_matrix[connected_id] = possible_distance
515                        predecessor[connected_id] = current_id
516                        if i == len_graph - 1:
517                            raise Exception(
518                                "Graph contains a negative weight cycle"
519                            )
520        # Check if destination is reachable
521        if distance_matrix[destination_id] == float("inf"):
522            raise Exception(
523                "Something went wrong, the origin and destination nodes are not connected."
524            )
525
526        return {
527            "path": self.__reconstruct_path__(destination_id, predecessor),
528            "length": distance_matrix[destination_id],
529        }

192        self,
193        origin_id: int | set[int],
194        destination_id: int,
195        max_edge_weight: int | float | None = None,
196    ) -> dict:
197        """
198        Function:
199
200        - Identify the shortest path between two nodes in a sparse network graph using Dijkstra's algorithm with buckets (Dial's algorithm)
201        - This is particularly efficient for graphs where most edge weights are >= 1 and the maximum edge weight is small
202        - This implementation safely supports non-integer weights
203        - Return a dictionary of various path information including:
204            - `path`: A list of node ids in the order they are visited
205            - `length`: The length of the path from the origin node to the destination node
206
207        Required Arguments:
208
209        - `origin_id`
210            - Type: int | set[int]
211            - What: The id(s) of the origin node(s) from the graph dictionary to start the shortest path from
212        - `destination_id`
213            - Type: int
214            - What: The id of the destination node from the graph dictionary to end the shortest path at
215
216        Optional Arguments:
217
218        - `max_edge_weight`
219            - Type: int | float | None
220            - What: The maximum edge weight in the graph. If None, it will be calculated.
221            - Default: None
222        """
223        # Input Validation
224        self.__input_check__(origin_id=origin_id, destination_id=destination_id)
225        origin_ids = {origin_id} if isinstance(origin_id, int) else origin_id
226
227        if max_edge_weight is None:
228            max_edge_weight = 0
229            for node_edges in self.graph:
230                if node_edges:
231                    m = max(node_edges.values())
232                    if m > max_edge_weight:
233                        max_edge_weight = m
234        max_edge_weight = math.ceil(max_edge_weight)
235
236        # Variable Initialization
237        distance_matrix = [float("inf")] * len(self.graph)
238        predecessor = [-1] * len(self.graph)
239        num_buckets = max_edge_weight + 1
240        buckets = [[] for _ in range(num_buckets)]
241
242        for oid in origin_ids:
243            distance_matrix[oid] = 0
244            buckets[0].append(oid)
245
246        current_dist = 0
247        nodes_in_buckets = len(origin_ids)
248
249        while nodes_in_buckets > 0:
250            bucket_idx = current_dist % num_buckets
251            while not buckets[bucket_idx]:
252                current_dist += 1
253                bucket_idx = current_dist % num_buckets
254                if nodes_in_buckets == 0:
255                    break
256                # If we've already found a path shorter than the current bucket minimum, we can exit
257                if distance_matrix[destination_id] < current_dist:
258                    break
259
260            if (
261                nodes_in_buckets == 0
262                or distance_matrix[destination_id] < current_dist
263            ):
264                break
265
266            current_id = buckets[bucket_idx].pop()
267            nodes_in_buckets -= 1
268
269            # Skip if we found a better path already (lazy removal)
270            if distance_matrix[current_id] < current_dist:
271                continue
272
273            # Note: We do not break immediately if current_id == destination_id
274            # because weights < 1 might allow a shorter path to be found within the same bucket.
275            # The loop terminates when current_dist > distance_matrix[destination_id].
276
277            for connected_id, connected_distance in self.graph[
278                current_id
279            ].items():
280                possible_distance = (
281                    distance_matrix[current_id] + connected_distance
282                )
283                if possible_distance < distance_matrix[connected_id]:
284                    distance_matrix[connected_id] = possible_distance
285                    predecessor[connected_id] = current_id
286                    buckets[int(possible_distance) % num_buckets].append(
287                        connected_id
288                    )
289                    nodes_in_buckets += 1
290
291        if distance_matrix[destination_id] == float("inf"):
292            raise Exception(
293                "Something went wrong, the origin and destination nodes are not connected."
294            )
295
296        return {
297            "path": self.__reconstruct_path__(destination_id, predecessor),
298            "length": distance_matrix[destination_id],
299        }
