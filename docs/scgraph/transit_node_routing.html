 86class TNRGraphPreprocessing:
 87    def __compute_access_nodes__(
 88        self, node_id: int, forward: bool = True
 89    ) -> dict[int, float]:
 90        """
 91        Function:
 92        - Perform a restricted upward CH search to find the first transit nodes encountered
 93        """
 94        access_nodes = {}
 95        distances = {node_id: 0}
 96        open_leaves = [(0, node_id)]
 97
 98        while open_leaves:
 99            dist, current_id = heappop(open_leaves)
100            if dist > distances.get(current_id, float("inf")):
101                continue
102
103            # If we hit a transit node, we stop searching this branch
104            if current_id in self.transit_nodes:
105                if dist < access_nodes.get(current_id, float("inf")):
106                    access_nodes[current_id] = dist
107                continue
108
109            # Check neighbors in the upward graph
110            # If current_id is an added node (rank inf), it uses original_graph logic
111            # (handled by inheriting from CHGraph)
112            neighbors = self.__get_neighbors__(current_id, forward)
113
114            for neighbor_id, weight in neighbors.items():
115                new_dist = dist + weight
116                if new_dist < distances.get(neighbor_id, float("inf")):
117                    distances[neighbor_id] = new_dist
118                    heappush(open_leaves, (new_dist, neighbor_id))
119        return access_nodes

121    def __preprocess_tnr__(self, num_transit_nodes: int):
122        """
123        Function:
124        - Perform TNR preprocessing on top of an existing CH
125        """
126        # 1. Select Transit Nodes (Top N by rank)
127        ranked_nodes = sorted(
128            range(self.nodes_count), key=lambda i: self.ranks[i], reverse=True
129        )
130        self.transit_nodes = set(ranked_nodes[:num_transit_nodes])
131
132        # 2. Compute Access Nodes for all nodes
133        self.forward_access_nodes = [
134            self.__compute_access_nodes__(i, True)
135            for i in range(self.nodes_count)
136        ]
137        self.backward_access_nodes = [
138            self.__compute_access_nodes__(i, False)
139            for i in range(self.nodes_count)
140        ]
141
142        # 3. Compute Distance Table between transit nodes
143        self.distance_table = {}
144        t_nodes_list = list(self.transit_nodes)
145
146        # Note: This is O(T^2) which is slow for Python.
147        # For a production implementation, we'd use many-to-one Dijkstra.
148        for origin in t_nodes_list:
149            # We use CH search for distance
150            # This can be optimized by using a simpler Dijkstra if we're only going transit-to-transit
151            for target in t_nodes_list:
152                if origin == target:
153                    self.distance_table[(origin, target)] = 0
154                    continue
155                # Bidirectional CH search is very fast
156                res = CHGraph.search(self, origin, target)
157                self.distance_table[(origin, target)] = res["length"]

161    def __local_search__(
162        self,
163        origin_id: int,
164        destination_id: int,
165        upper_bound: float,
166        length_only: bool,
167    ) -> Optional[dict]:
168        forward_distances = {origin_id: 0}
169        forward_parent = {origin_id: -1}
170        forward_open_leaves = [(0, origin_id)]
171
172        backward_distances = {destination_id: 0}
173        backward_parent = {destination_id: -1}
174        backward_open_leaves = [(0, destination_id)]
175
176        best_dist = upper_bound
177        meeting_node = -1
178
179        while forward_open_leaves or backward_open_leaves:
180            if forward_open_leaves:
181                current_distance, current_id = heappop(forward_open_leaves)
182                if current_distance > best_dist:
183                    forward_open_leaves = []
184                elif current_id not in self.transit_nodes:
185                    current_rank = self.__get_rank__(current_id)
186                    neighbors = (
187                        self.forward_graph[current_id]
188                        if current_id < self.nodes_count
189                        else self.original_graph[current_id]
190                    )
191
192                    for neighbor_id, weight in neighbors.items():
193                        if (
194                            self.__get_rank__(neighbor_id) <= current_rank
195                            and neighbor_id < self.nodes_count
196                        ):
197                            continue
198                        new_dist = current_distance + weight
199                        if new_dist < forward_distances.get(
200                            neighbor_id, float("inf")
201                        ):
202                            forward_distances[neighbor_id] = new_dist
203                            if not length_only:
204                                forward_parent[neighbor_id] = current_id
205                            heappush(
206                                forward_open_leaves, (new_dist, neighbor_id)
207                            )
208                            if (
209                                neighbor_id in backward_distances
210                                and new_dist + backward_distances[neighbor_id]
211                                < best_dist
212                            ):
213                                best_dist = (
214                                    new_dist + backward_distances[neighbor_id]
215                                )
216                                meeting_node = neighbor_id
217
218            if backward_open_leaves:
219                current_distance, current_id = heappop(backward_open_leaves)
220                if current_distance > best_dist:
221                    backward_open_leaves = []
222                elif current_id not in self.transit_nodes:
223                    current_rank = self.__get_rank__(current_id)
224                    neighbors = (
225                        self.backward_graph[current_id]
226                        if current_id < self.nodes_count
227                        else self.original_graph[current_id]
228                    )
229
230                    for neighbor_id, weight in neighbors.items():
231                        if (
232                            self.__get_rank__(neighbor_id) <= current_rank
233                            and neighbor_id < self.nodes_count
234                        ):
235                            continue
236                        new_dist = current_distance + weight
237                        if new_dist < backward_distances.get(
238                            neighbor_id, float("inf")
239                        ):
240                            backward_distances[neighbor_id] = new_dist
241                            if not length_only:
242                                backward_parent[neighbor_id] = current_id
243                            heappush(
244                                backward_open_leaves, (new_dist, neighbor_id)
245                            )
246                            if (
247                                neighbor_id in forward_distances
248                                and new_dist + forward_distances[neighbor_id]
249                                < best_dist
250                            ):
251                                best_dist = (
252                                    new_dist + forward_distances[neighbor_id]
253                                )
254                                meeting_node = neighbor_id
255
256            forward_min = (
257                forward_open_leaves[0][0]
258                if forward_open_leaves
259                else float("inf")
260            )
261            backward_min = (
262                backward_open_leaves[0][0]
263                if backward_open_leaves
264                else float("inf")
265            )
266            if forward_min > best_dist and backward_min > best_dist:
267                break
268
269        if length_only:
270            return {"length": best_dist}
271
272        if meeting_node != -1:
273            path = self.__reconstruct_ch_path__(
274                origin_id,
275                destination_id,
276                meeting_node,
277                forward_parent,
278                backward_parent,
279            )
280            return {"path": path, "length": best_dist}
281
282        return None

284    def search(
285        self,
286        origin_id: int,
287        destination_id: int,
288        length_only: bool = False,
289        **kwargs,
290    ) -> dict[str, Any]:
291        """
292        Function:
293        - Perform a TNR search with a highly pruned local CH fallback
294        """
295        if origin_id == destination_id:
296            return (
297                {"length": 0}
298                if length_only
299                else {"path": [origin_id], "length": 0}
300            )
301
302        # Access Nodes for origin/destination
303        if origin_id < self.nodes_count:
304            f_access = self.forward_access_nodes[origin_id]
305        else:
306            f_access = self.__compute_access_nodes__(origin_id, True)
307
308        if destination_id < self.nodes_count:
309            b_access = self.backward_access_nodes[destination_id]
310        else:
311            b_access = self.__compute_access_nodes__(destination_id, False)
312
313        best_dist = float("inf")
314
315        # Global Query via Distance Table
316        for t_f, d_f in f_access.items():
317            for t_b, d_b in b_access.items():
318                d_table = self.distance_table.get((t_f, t_b), float("inf"))
319                total = d_f + d_table + d_b
320                if total < best_dist:
321                    best_dist = total
322
323        # Locality Filter / Correctness Check
324        if length_only:
325            return self.__local_search__(
326                origin_id, destination_id, best_dist, True
327            )
328
329        local_res = self.__local_search__(
330            origin_id, destination_id, best_dist, False
331        )
332        if local_res is not None:
333            return local_res
334
335        # Reconstructing path from transit pairs is complex without unpacking tables.
336        # Fall back to a standard CH search if global TNR path is needed.
337        return CHGraph.search(self, origin_id, destination_id)

340class TNRGraph(TNRGraphIO, TNRGraphPreprocessing, TNRGraphAlgorithms, CHGraph):
341    def __init__(
342        self,
343        graph: Optional[list[dict[int, int | float]]] = None,
344        num_transit_nodes: int = 100,
345        **kwargs,
346    ):
347        """
348        Function:
349        - Create a Transit Node Routing Graph object
350
351        Optional Arguments:
352        - `graph`: Original graph adjacency list
353        - `num_transit_nodes`: Number of top-ranked nodes to use as transit nodes
354        - `kwargs`: Used when loading from saved state
355        """
356        if "transit_nodes" in kwargs:
357            # Loading from saved state
358            self.nodes_count = kwargs.get("nodes_count")
359            self.transit_nodes = set(kwargs.get("transit_nodes"))
360
361            # Reconstruct distance_table (keys are strings like "(1,2)")
362            self.distance_table = {}
363            for k, v in kwargs.get("distance_table").items():
364                pair = tuple(map(int, k.strip("()").split(",")))
365                self.distance_table[pair] = v
366
367            # Reconstruct access nodes
368            self.forward_access_nodes = [
369                {int(k): v for k, v in d.items()}
370                for d in kwargs.get("forward_access_nodes")
371            ]
372            self.backward_access_nodes = [
373                {int(k): v for k, v in d.items()}
374                for d in kwargs.get("backward_access_nodes")
375            ]
376
377            # Initialize CH part
378            ch_data = kwargs.get("ch_data")
379            # Convert graph keys to integers
380            for key in ["forward_graph", "backward_graph", "original_graph"]:
381                if ch_data.get(key) is not None:
382                    ch_data[key] = [
383                        {int(k): v for k, v in d.items()} for d in ch_data[key]
384                    ]
385            super().__init__(**ch_data)
386        else:
387            # Preprocess from original graph
388            super().__init__(graph=graph)
389            self.__preprocess_tnr__(num_transit_nodes)
390
391    def get_shortest_path(
392        self, origin_id: int, destination_id: int, **kwargs: Any
393    ) -> dict[str, Any]:
394        return self.search(origin_id, destination_id)
