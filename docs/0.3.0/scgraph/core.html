160    def dijkstra(self, origin: str, destination: str):
161        """
162        Function:
163
164        - Identify the minimum length route between two nodes in a sparse network graph
165        - Return that route as a list of node keys in the order they are visited as well as the total distance traveled
166
167        Algorithm Notes:
168
169        - Runs in O(n^2) time
170
171        Required Arguments:
172
173        - `origin`
174            - Type: str
175            - What: The origin node key
176        - `destination`
177            - Type: str
178            - What: The destination node key
179
180        Optional Arguments:
181
182        - None
183        """
184        ids = list(
185            set(
186                [key for key in self.graph.keys()]
187                + [
188                    key
189                    for subdict in self.graph.values()
190                    for key in subdict.keys()
191                ]
192            )
193        )
194        id_map = {key: idx for idx, key in enumerate(ids)}
195        id_map_inv = {v: k for k, v in id_map.items()}
196
197        origin_id = id_map[origin]
198        destination_id = id_map[destination]
199
200        id_graph = {
201            id_map[origin]: {
202                id_map[destination]: distance
203                for destination, distance in self.graph[origin].items()
204            }
205            for origin in self.graph.keys()
206        }
207
208        distance_matrix = [float("inf") for i in id_map]
209        branch_tip_distances = [float("inf") for i in id_map]
210        predecessor = [None for i in id_map]
211
212        distance_matrix[origin_id] = 0
213        branch_tip_distances[origin_id] = 0
214
215        while True:
216            current_distance = min(branch_tip_distances)
217            if current_distance == float("inf"):
218                return None
219            current = branch_tip_distances.index(current_distance)
220            if current == destination_id:
221                break
222            for destination, distance in id_graph[current].items():
223                new_distance = current_distance + distance
224                if new_distance < distance_matrix[destination]:
225                    distance_matrix[destination] = new_distance
226                    predecessor[destination] = current
227                    branch_tip_distances[destination] = new_distance
228            branch_tip_distances[current] = float("inf")
229
230        id_path = [current]
231        while predecessor[current] is not None:
232            current = predecessor[current]
233            id_path.append(current)
234
235        output = {
236            "path_ids": [id_map_inv[id] for id in id_path[::-1]],
237            "length": distance_matrix[destination_id],
238        }
239        if len(self.nodes) > 0:
240            output["path"] = [self.nodes[id] for id in output["path_ids"]]
241        return output

248        return_id_path: bool = True,
249        return_node_path: bool = True,
250    ):
251        """
252        Function:
253
254        - Identify the minimum length route between two nodes in a sparse network graph
255        - Return that route as a list of node keys in the order they are visited
256
257        Algorithm Notes:
258
259        - Runs between O(n) time and O(n^2) time depending on the sparsity of the graph
260        - This algorithm is faster than `dijkstra` for graphs that terminate branches in dead ends
261        - In general, this applies to graphs that are more sparse
262        - For dense graphs, this algorithm will be slower than `dijkstra` because of the overhead of maintaining the open leaves as a dictionary
263
264        Required Arguments:
265
266        - `origin`
267            - Type: str
268            - What: The origin node key
269        - `destination`
270            - Type: str
271            - What: The destination node key
272
273        Optional Arguments:
274
275        - None
276        """
277        ids = list(
278            set(
279                [key for key in self.graph.keys()]
280                + [
281                    key
282                    for subdict in self.graph.values()
283                    for key in subdict.keys()
284                ]
285            )
286        )
287        id_map = {key: idx for idx, key in enumerate(ids)}
288        id_map_inv = {v: k for k, v in id_map.items()}
289
290        origin_id = id_map[origin]
291        destination_id = id_map[destination]
292
293        id_graph = {
294            id_map[origin]: {
295                id_map[destination]: distance
296                for destination, distance in self.graph[origin].items()
297            }
298            for origin in self.graph.keys()
299        }
300
301        distance_matrix = [float("inf") for i in id_map]
302        open_leaves = {}
303        predecessor = [None for i in id_map]
304
305        distance_matrix[origin_id] = 0
306        open_leaves[origin_id] = 0
307
308        while True:
309            if len(open_leaves) == 0:
310                return None
311            current = min(open_leaves, key=open_leaves.get)
312            open_leaves.pop(current)
313            if current == destination_id:
314                break
315            current_distance = distance_matrix[current]
316            for destination, distance in id_graph[current].items():
317                if current_distance + distance < distance_matrix[destination]:
318                    distance_matrix[destination] = current_distance + distance
319                    predecessor[destination] = current
320                    open_leaves[destination] = current_distance + distance
321
322        id_path = [current]
323        while predecessor[current] is not None:
324            current = predecessor[current]
325            id_path.append(current)
326
327        output = {
328            "path_ids": [id_map_inv[id] for id in id_path[::-1]],
329            "length": distance_matrix[destination_id],
330        }
331        if len(self.nodes) > 0:
332            output["path"] = [self.nodes[id] for id in output["path_ids"]]
333        return output

498        - Add a node to the network
499        - Return None
500
501        Required Arguments:
502
503        - `node`
504            - Type: dict
505            - What: A dictionary with the keys 'latitude' and 'longitude'
506        - `name`
507            - Type: str
508            - What: The name of the node
509
510        Optional Arguments:
511
512        - `circuity`
513            - Type: float | int
514            - What: The circuity of the network
515            - Default: 2
516        - `distance_calculation`
517            - Type: str
518            - What: The distance calculation to use
519            - Default: 'haversine'
520            - Options: 'haversine'
521        - `node_addition_type`
522            - Type: str
523            - What: The type of node addition to use
524            - Default: 'quadrant'
525            - Options:
526                - 'quadrant': Add the closest node in each quadrant (ne, nw, se, sw) to the distance matrix for this node
527                - 'closest': Add only the closest node to the distance matrix for this node
528                - 'all': Add all nodes to the distance matrix for this node
529            - Notes:
530                - `dijkstra_v2` will operate substantially faster if the `node_addition_type` is set to 'quadrant' or 'closest'
531                - `dijkstra` will operate at the similar speeds regardless of the `node_addition_type`
532                - When using `all`, you should consider using `dijkstra` instead of `dijkstra_v2` as it will be faster
533
534        """
535        # Validate the node
536        assert isinstance(name, str), f"Your node name ({name}) is not a string"
537        assert isinstance(
538            node, dict
539        ), f"Your node ({name}) must be a dictionary"
540        assert (
541            "latitude" in node.keys()

556        assert (
557            circuity >= 1
558        ), f"Your circuity for node  ({name}) must be greater than or equal to 1"
559        # Validate the distance calculation
560        assert distance_calculation in [
561            "haversine"
562        ], f"Invalid distance calculation provided for node ({name}) with option ({distance_calculation}), valid options are: haversine"
563        assert node_addition_type in [
564            "quadrant",
565            "all",
566            "closest",
567        ], f"Invalid node addition type provided for node ({name}) with option ({node_addition_type}), valid options are: quadrant, all, closest"
568        if node_addition_type == "quadrant":

570            # only the closest node in each directional quadrant (ne, nw, se, sw) to the graph

602                self.graph[name][node_i_id] = distance
603            # Add the node to the nodes dictionary

605        elif node_addition_type == "closest":

607            # only the closest node to the graph
608            closest_node = None
609            closest_distance = float("inf")
610            for node_i_id, node_i in self.nodes.items():
611                distance = Distance.haversine(node, node_i, circuity=circuity)
612                if distance < closest_distance:

614                    closest_distance = distance
615            self.graph[name] = {closest_node: closest_distance}
616            self.graph[closest_node][name] = closest_distance
617            # Add the node to the nodes dictionary
618            self.nodes[name] = node
619        else:
620            raise ValueError(
621                f"Invalid node addition type: {node_addition_type}"

161    def dijkstra(self, origin: str, destination: str):
162        """
163        Function:
164
165        - Identify the minimum length route between two nodes in a sparse network graph
166        - Return that route as a list of node keys in the order they are visited as well as the total distance traveled
167
168        Algorithm Notes:
169
170        - Runs in O(n^2) time
171
172        Required Arguments:
173
174        - `origin`
175            - Type: str
176            - What: The origin node key
177        - `destination`
178            - Type: str
179            - What: The destination node key
180
181        Optional Arguments:
182
183        - None
184        """
185        ids = list(
186            set(
187                [key for key in self.graph.keys()]
188                + [
189                    key
190                    for subdict in self.graph.values()
191                    for key in subdict.keys()
192                ]
193            )
194        )
195        id_map = {key: idx for idx, key in enumerate(ids)}
196        id_map_inv = {v: k for k, v in id_map.items()}
197
198        origin_id = id_map[origin]
199        destination_id = id_map[destination]
200
201        id_graph = {
202            id_map[origin]: {
203                id_map[destination]: distance
204                for destination, distance in self.graph[origin].items()
205            }
206            for origin in self.graph.keys()
207        }
208
209        distance_matrix = [float("inf") for i in id_map]
210        branch_tip_distances = [float("inf") for i in id_map]
211        predecessor = [None for i in id_map]
212
213        distance_matrix[origin_id] = 0
214        branch_tip_distances[origin_id] = 0
215
216        while True:
217            current_distance = min(branch_tip_distances)
218            if current_distance == float("inf"):
219                return None
220            current = branch_tip_distances.index(current_distance)
221            if current == destination_id:
222                break
223            for destination, distance in id_graph[current].items():
224                new_distance = current_distance + distance
225                if new_distance < distance_matrix[destination]:
226                    distance_matrix[destination] = new_distance
227                    predecessor[destination] = current
228                    branch_tip_distances[destination] = new_distance
229            branch_tip_distances[current] = float("inf")
230
231        id_path = [current]
232        while predecessor[current] is not None:
233            current = predecessor[current]
234            id_path.append(current)
235
236        output = {
237            "path_ids": [id_map_inv[id] for id in id_path[::-1]],
238            "length": distance_matrix[destination_id],
239        }
240        if len(self.nodes) > 0:
241            output["path"] = [self.nodes[id] for id in output["path_ids"]]
242        return output
