 81            - Note: All nodes must be included as origins in the graph regardless of if they have any connected destinations
 82
 83        Optional Arguments:
 84
 85        - None
 86        """
 87        origin_id = 0
 88        destination_id = len(graph) + 1
 89
 90        distance_matrix = [float("inf") for i in graph]
 91        open_leaves = {}
 92        predecessor = [None for i in graph]

 95        open_leaves[origin_id] = 0

 98            if len(open_leaves) == 0:

154    def dijkstra(
155        graph: list[dict], origin_id: int, destination_id: int
156    ) -> dict:
157        """
158        Function:
159
160        - Identify the shortest path between two nodes in a sparse network graph using a modified dijkstra algorithm
161            - Modifications allow for a sparse distance matrix to be used instead of a dense distance matrix
162            - This can dramatically reduce the memory and compute requirements of the algorithm
163            - This algorithm runs in O(n^2) time where n is the number of nodes in the graph
164        - Return a dictionary of various path information including:
165            - `id_path`: A list of node ids in the order they are visited
166            - `path`: A list of node dictionaries (lat + long) in the order they are visited
167
168        Required Arguments:
169
170        - `graph`
171            - Type: list[dict]
172            - What: A list of dictionaries where the indicies are origin node ids and the values are dictionaries of destination node ids and distances
173        - `origin_id`
174            - Type: int
175            - What: The id of the origin node from the graph dictionary to start the shortest path from
176        - `destination_id`
177            - Type: int
178            - What: The id of the destination node from the graph dictionary to end the shortest path at
179
180        Optional Arguments:
181
182        - None
183        """
184        Graph.input_check(
185            graph=graph, origin_id=origin_id, destination_id=destination_id
186        )
187        distance_matrix = [float("inf") for i in graph]
188        branch_tip_distances = [float("inf") for i in graph]
189        predecessor = [None for i in graph]
190
191        distance_matrix[origin_id] = 0
192        branch_tip_distances[origin_id] = 0
193
194        while True:
195            current_distance = min(branch_tip_distances)
196            if current_distance == float("inf"):
197                raise Exception(
198                    "Something went wrong, the origin and destination nodes are not connected."
199                )
200            current_id = branch_tip_distances.index(current_distance)
201            branch_tip_distances[current_id] = float("inf")
202            if current_id == destination_id:
203                break
204            for connected_id, connected_distance in graph[current_id].items():
205                possible_distance = current_distance + connected_distance
206                if possible_distance < distance_matrix[connected_id]:
207                    distance_matrix[connected_id] = possible_distance
208                    predecessor[connected_id] = current_id
209                    branch_tip_distances[connected_id] = possible_distance
210
211        output_path = [current_id]
212        while predecessor[current_id] is not None:
213            current_id = predecessor[current_id]
214            output_path.append(current_id)
215
216        output_path.reverse()
217
218        return {
219            "path": output_path,
220            "length": hard_round(4, distance_matrix[destination_id]),
221        }

234            - This can dramatically reduce the memory and compute requirements of the algorithm

236                - Where n is the number of nodes in the graph
237            - For dense graphs, this algorithm runs in O(n^2) time
238                - Where n is the number of nodes in the graph

240            - `id_path`: A list of node ids in the order they are visited

242

244
245        - `graph`
246            - Type: list[dict]
247            - What: A list of dictionaries where the indicies are origin node ids and the values are dictionaries of destination node ids and distances
248        - `origin_id`
249            - Type: int
250            - What: The id of the origin node from the graph dictionary to start the shortest path from
251        - `destination_id`
252            - Type: int
253            - What: The id of the destination node from the graph dictionary to end the shortest path at
254
255        Optional Arguments:
256
257        - None
258        """
259        Graph.input_check(
260            graph=graph, origin_id=origin_id, destination_id=destination_id
261        )
262        distance_matrix = [float("inf") for i in graph]
263        open_leaves = {}
264        predecessor = [None for i in graph]
265
266        distance_matrix[origin_id] = 0
267        open_leaves[origin_id] = 0
268
269        while True:
270            if len(open_leaves) == 0:
271                raise Exception(
272                    "Something went wrong, the origin and destination nodes are not connected."
273                )

277                break

282                    distance_matrix[connected_id] = possible_distance
283                    predecessor[connected_id] = current_id

288            current_id = predecessor[current_id]

398    def get_shortest_path(
399        self,
400        origin_node,
401        destination_node,
402        output_units: str = "km",
403        algorithm_fn=Graph.dijkstra_makowski,
404        off_graph_circuity: [float, int] = 1,
405        node_addition_type: str = "quadrant",
406        node_addition_circuity: [float, int] = 4,
407        geograph_units: str = "km",
408        output_coordinate_path: str = "list_of_lists",
409        output_path: bool = False,
410        **kwargs,
411    ) -> dict:
412        """
413        Function:
414
415        - Identify the shortest path between two nodes in a sparse network graph
416
417        - Return a dictionary of various path information including:
418            - `id_path`: A list of node ids in the order they are visited
419            - `path`: A list of node dictionaries (lat + long) in the order they are visited
420            - `length`: The length of the path
421
422         Required Arguments:
423
424        - `origin_node`
425            - Type: dict
426            - What: A dictionary with the keys 'latitude' and 'longitude'
427        - `destination_node`
428            - Type: dict
429            - What: A dictionary with the keys 'latitude' and 'longitude'
430
431        Optional Arguments:
432
433        - `output_units`
434            - Type: str
435            - What: The units in which to return the length of the path
436            - Default: 'km'
437            - Options:
438                - 'km': Kilometers
439                - 'm': Meters
440                - 'mi': Miles
441                - 'ft': Feet
442        - `algorithm_fn`
443            - Type: function | method
444            - What: The algorithm function to identify the shortest path
445            - Default: 'Graph.dijkstra_makowski'
446            - Options:
447                - 'Graph.dijkstra': A modified dijkstra algorithm that uses a sparse distance matrix to identify the shortest path
448                - 'Graph.dijkstra_makowski': A modified dijkstra algorithm that uses a sparse distance matrix to identify the shortest path
449                - Any user defined algorithm that takes the arguments:
450                    - `graph`: A dictionary of dictionaries where the keys are origin node ids and the values are dictionaries of destination node ids and distances
451                    - `origin`: The id of the origin node from the graph dictionary to start the shortest path from
452                    - `destination`: The id of the destination node from the graph dictionary to end the shortest path at
453        - `off_graph_circuity`
454            - Type: float | int
455            - What: The circuity factor to apply to any distance calculations between your origin and destination nodes and their connecting nodes in the graph
456            - Default: 1
457            - Notes:
458                - For alogrithmic solving purposes, the node_addition_circuity is applied to the origin and destination nodes when they are added to the graph
459                - This is only applied after an `optimal solution` using the `node_addition_circuity` has been found when it is then adjusted to equal the `off_graph_circuity`
460        - `node_addition_type`
461            - Type: str
462            - What: The type of node addition to use when adding your origin node to the distance matrix
463            - Default: 'quadrant'
464            - Options:
465                - 'quadrant': Add the closest node in each quadrant (ne, nw, se, sw) to the distance matrix for this node
466                - 'closest': Add only the closest node to the distance matrix for this node
467                - 'all': Add all nodes to the distance matrix for this node
468            - Notes:
469                - `dijkstra_makowski` will operate substantially faster if the `node_addition_type` is set to 'quadrant' or 'closest'
470                - `dijkstra` will operate at the similar speeds regardless of the `node_addition_type`
471                - When using `all`, you should consider using `dijkstra` instead of `dijkstra_makowski` as it will be faster
472                - The destination node is always added as 'all' regardless of the `node_addition_type` setting
473                    - This guarantees that any destination node will be connected to any origin node regardless of how or where the origin node is added to the graph
474                - If the passed graph is not a connected graph (meaning it is comprised of multiple disconnected networks)
475                    - The entrypoints generated using the `node_addition_type` will determine which disconnected networks will be used to calculate the `optimal route`
476        - `node_addition_circuity`
477            - Type: float | int
478            - What: The circuity factor to apply when adding your origin and destination nodes to the distance matrix
479            - Default: 4
480            - Note:
481                - This defaults to 4 to prevent the algorithm from taking a direct route in direction of the destination over some impassible terrain (EG: a maritime network that goes through land)
482                - A higher value will push the algorithm to join the network at a closer node to avoid the extra distance from the circuity factor
483                - This is only relevant if `node_addition_type` is set to 'quadrant' or 'all' as it affects the choice on where to enter the graph network
484                - This factor is used to calculate the node sequence for the `optimal route`, however the reported `length` of the path will be calculated using the `off_graph_circuity` factor

500                - 'list_of_dicts': A list of dictionaries with keys 'latitude' and 'longitude'

503            - Type: bool
504            - What: Whether to output the path as a list of geograph node ids (for debugging and other advanced uses)
505            - Default: False
506        - `**kwargs`
507            - Additional keyword arguments. These are included for forwards and backwards compatibility reasons, but are not currently used.
508        """
509        original_graph_length = len(self.graph)
510        # Add the origin and destination nodes to the graph
511        origin_id = self.add_node(
512            node=origin_node,
513            node_addition_type=node_addition_type,
514            circuity=node_addition_circuity,
515        )
516        destination_id = self.add_node(
517            node=destination_node,
518            node_addition_type="all",
519            circuity=node_addition_circuity,
520        )
521
522        try:
523            output = algorithm_fn(
524                graph=self.graph,
525                origin_id=origin_id,
526                destination_id=destination_id,
527            )
528            output["coordinate_path"] = self.get_coordinate_path(output["path"])
529            output["length"] = self.adujust_circuity_length(
530                output=output,
531                node_addition_circuity=node_addition_circuity,
532                off_graph_circuity=off_graph_circuity,
533            )
534            output["length"] = distance_converter(
535                output["length"],
536                input_units=geograph_units,
537                output_units=output_units,
538            )
539            if output_coordinate_path == "list_of_dicts":
540                output["coordinate_path"] = [
541                    {"latitude": i[0], "longitude": i[1]}
542                    for i in output["coordinate_path"]
543                ]
544            if not output_path:
545                del output["path"]
546            while len(self.graph) > original_graph_length:
547                self.remove_appended_node()
548            return output
549        except Exception as e:
550            while len(self.graph) > original_graph_length:
551                self.remove_appended_node()
552            raise e

554    def adujust_circuity_length(
555        self,
556        output: dict,
557        node_addition_circuity: [float, int],
558        off_graph_circuity: [float, int],
559    ) -> [float, int]:
560        """
561        Function:
562
563        - Adjust the length of the path to account for the circuity factors applied to the origin and destination nodes
564
565        Required Arguments:
566
567        - `output`
568            - Type: dict
569            - What: The output from the algorithm function
570        - `node_addition_circuity`
571            - Type: float | int
572            - What: The circuity factor that was applied when adding your origin and destination nodes to the distance matrix
573        - `off_graph_circuity`
574            - Type: float | int
575            - What: The circuity factor to apply to any distance calculations between your origin and destination nodes and their connecting nodes in the graph
576        """
577        coordinate_path = output["coordinate_path"]
578        # If the path does not enter the graph, undo the node_addition_circuity and apply the off_graph_circuity
579        if len(output["coordinate_path"]) == 2:
580            return (
581                output["length"] / node_addition_circuity
582            ) * off_graph_circuity
583        else:
584            direct_off_graph_length = haversine(
585                coordinate_path[0], coordinate_path[1], circuity=1
586            ) + haversine(coordinate_path[-2], coordinate_path[-1], circuity=1)
587            return round(
588                output["length"]
589                + direct_off_graph_length
590                * (off_graph_circuity - node_addition_circuity),
591                4,
592            )

687            "closest",
688        ], f"Invalid node addition type provided ({node_addition_type}), valid options are: ['quadrant', 'all', 'closest']"
